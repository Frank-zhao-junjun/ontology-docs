1. 文档概述
1.1 文档目的
本规格说明书定义"本体模型+AI驱动的原生应用系统"的完整需求，该系统是一个通用企业级应用框架，通过四大元模型（数据、行为、规则、流程）构建业务本体抽象，让AI贯穿"生成期（Auto-Dev）"与"运行期（Runtime）"全生命周期。
1.2 目标读者
系统架构师
后端/前端工程师
AI/算法工程师
产品经理
领域专家
1.3 术语表
表格
术语	定义
Metamodel（元模型）	描述业务本体结构的模型，包含数据、行为、规则、流程四类
Ontology（本体）	领域概念的显式规范，包括实体、属性、关系、约束
Skill（领域技能）	后端封装的业务能力单元，对应特定领域操作
Agent（智能编排器）	运行时决策中枢，集成意图分析、LLM策略、工具调用
Hybrid UI（混合交互）	自然语言界面（NLI）与结构化UI（GUI）的双模融合
Auto-Dev（生成期）	基于元模型自动生成代码的开发阶段
Semantic Injection（语义注入）	将元模型定义注入AI系统Prompt的机制

3.3.3 运行时状态机引擎
Python
复制
class SimpleStateMachine:
    def can_transition(self, entity, transition_id: str) -> bool:
        current_state = entity.status
        transition = self.get_transition(transition_id)
        
        # 检查当前状态是否匹配
        if current_state not in transition.from_states:
            return False
            
        # 检查规则条件（引用规则模型）
        for rule_ref in transition.pre_conditions:
            if not self.rule_engine.evaluate(rule_ref, entity):
                return False
                
        return True
    
    def execute_transition(self, entity, transition_id: str, user: User):
        if not self.can_transition(entity, transition_id):
            raise InvalidTransitionError()
            
        transition = self.get_transition(transition_id)
        
        # 执行前置动作
        self.execute_actions(transition.pre_actions, entity)
        
        # 更新状态
        old_state = entity.status
        entity.status = transition.to
        entity.save()
        
        # 执行后置动作
        self.execute_actions(transition.post_actions, entity)
        
        # 记录状态变更（即使Q4选择无审计，状态变更本身需要追踪用于业务逻辑）
        self.log_state_change(entity, old_state, transition.to, user)

复制
# rule-model.yaml
RuleModel:
  rules:
    # 1. 字段级校验
    - id: "contract_no_format"
      name: "合同编号格式"
      type: "field_validation"
      entity: "contract"
      field: "contract_no"
      condition:
        type: "regex"
        pattern: "^HT-[0-9]{4}-[0-9]{6}$"
      error_message: "合同编号格式应为HT-YYYY-NNNNNN"
      
    - id: "amount_positive"
      name: "金额必须为正数"
      type: "field_validation"
      entity: "contract"
      field: "total_amount"
      condition:
        type: "range"
        min: 0
        exclusive_min: true
      error_message: "合同金额必须大于0"
      
    # 2. 跨字段校验
    - id: "date_logic"
      name: "日期逻辑"
      type: "cross_field_validation"
      entity: "contract"
      condition:
        type: "expression"
        expression: "end_date > start_date"
        fields: ["start_date", "end_date"]
      error_message: "结束日期必须晚于开始日期"
      
    # 3. 跨实体校验
    - id: "customer_active"
      name: "客户必须激活"
      type: "cross_entity_validation"
      entity: "contract"
      condition:
        type: "reference_check"
        field: "customer_id"
        ref_entity: "customer"
        ref_field: "status"
        ref_value: "active"
      error_message: "只能与状态为激活的客户签署合同"
      
    # 4. 聚合校验
    - id: "amount_match"
      name: "金额匹配"
      type: "aggregation_validation"
      entity: "contract"
      condition:
        type: "sum_match"
        master_field: "total_amount"
        detail_entity: "payment_schedule"
        detail_field: "amount"
        detail_foreign_key: "contract_id"
      error_message: "合同总金额必须等于各期收款金额之和"
      
    # 5. 时序规则
    - id: "first_invoice_timing"
      name: "首票时限"
      type: "temporal_rule"
      entity: "contract"
      trigger:
        event: "state_change"
        from: "approved"
        to: "active"
      condition:
        type: "deadline"
        deadline_field: "sign_date"
        days_after: 30
        check_entity: "invoice"
        check_condition: "exists with type='first' and contract_id={{entity.id}}"

3.4.3 规则引擎执行
Python
复制
class RuleEngine:
    def evaluate(self, rule_id: str, context: Dict) -> ValidationResult:
        rule = self.load_rule_from_db(rule_id)
        
        evaluator = self.get_evaluator(rule.type)
        return evaluator.evaluate(rule, context)
    
    def validate_entity(self, entity, action: str = None) -> List[ValidationResult]:
        """验证实体是否符合所有适用规则"""
        applicable_rules = self.get_rules_for_entity(entity.__class__.__name__, action)
        
        results = []
        for rule in applicable_rules:
            result = self.evaluate(rule.id, {"entity": entity})
            results.append(result)
            
        return results
3.5 流程模型（Process Model）

class SelfHealingExecutor:
    MAX_RETRIES = 2
    
    def execute_with_healing(self, tool_call: Dict, context: Dict) -> Result:
        attempt = 0
        last_error = None
        
        while attempt <= self.MAX_RETRIES:
            try:
                result = self.execute_tool(tool_call)
                return Success(result)
                
            except ExecutionError as e:
                last_error = e
                attempt += 1
                
                if attempt > self.MAX_RETRIES:
                    break
                
                # 构建修正Prompt
                healing_prompt = self._build_healing_prompt(
                    original_call=tool_call,
                    error_message=str(e),
                    error_type=e.type,
                    context=context
                )
                
                # 请求LLM生成修正方案
                correction = self.llm.generate_correction(healing_prompt)
                
                # 更新tool_call参数
                tool_call = self._apply_correction(tool_call, correction)
        
        # 重试耗尽，返回错误
        return Failure(
            message="执行失败，已达到最大重试次数",
            last_error=str(last_error),
            attempts=attempt
        )
    
    def _build_healing_prompt(self, original_call, error_message, error_type, context):
        return f"""
执行工具调用时发生错误，请分析并生成修正方案。

原始调用：
{json.dumps(original_call, indent=2)}

错误信息：{error_message}
错误类型：{error_type}

当前数据模型上下文：
{context['data_model']}

请分析错误原因，并返回修正后的工具调用参数。
如果是SQL错误（如字段不存在），请检查数据模型定义并修正字段名。
如果是参数错误，请修正参数格式或值。

# Skill接口规范
class DomainSkill(ABC):
    @property
    @abstractmethod
    def name(self) -> str: pass
    
    @property
    @abstractmethod
    def parameters_schema(self) -> Dict: pass
    
    @abstractmethod
    def execute(self, parameters: Dict, user: User) -> SkillResult: pass
    
    @property
    def required_permissions(self) -> List[str]:
        return []

# 示例：合同服务
class ContractService(DomainSkill):
    name = "contract_service"
    
    def create(self, parameters: Dict, user: User) -> SkillResult:
        """创建合同"""
        # 1. 参数校验（Pydantic模型）
        data = ContractCreateSchema(**parameters)
        
        # 2. 规则校验（调用规则引擎）
        validation_results = rule_engine.validate_entity(data, action="create")
        if not all(r.is_valid for r in validation_results):
            return SkillResult.error(validation_results)
        
        # 3. 状态机校验（初始状态）
        if data.status != "draft":
            return SkillResult.error("新合同必须是草稿状态")
        
        # 4. 持久化
        contract = Contract(**data.dict())
        contract.save()
        
        # 5. 返回结果
        return SkillResult.success({
            "id": contract.id,
            "message": f"合同 {contract.contract_no} 创建成功"
        })
    
    def transition(self, parameters: Dict, user: User) -> SkillResult:
        """状态流转"""
        contract_id = parameters.get("contract_id")
        transition_id = parameters.get("transition_id")
        
        contract = Contract.get_by_id(contract_id)
        
        # 调用状态机引擎
        try:
            state_machine.execute_transition(contract, transition_id, user)
            return SkillResult.success({
                "new_status": contract.status,
                "message": f"合同状态已变更为 {contract.status}"
            })
        except InvalidTransitionError as e:
            return SkillResult.error(str(e))

        # 1. 加载四大元模型
        data_model = self._load_data_model()
        behavior_model = self._load_behavior_model()
        rule_model = self._load_rule_model()
        process_model = self._load_process_model()
        
        # 2. 获取会话上下文
        context = self._get_conversation_context(conversation_id)
        
        # 3. 渲染Prompt模板
        template = self._get_template()
        return template.render(
            data_model=json.dumps(data_model, ensure_ascii=False),
            behavior_model=json.dumps(behavior_model, ensure_ascii=False),
            rule_model=json.dumps(rule_model, ensure_ascii=False),
            process_model=json.dumps(process_model, ensure_ascii=False),
            user=context['user'],
            conversation_history=context['history'],
            focus_entity=context['focus_entity']
        )

class SQLSecurityFilter:
    FORBIDDEN_KEYWORDS = [
        'INSERT', 'UPDATE', 'DELETE', 'DROP', 'CREATE', 'ALTER',
        'TRUNCATE', 'EXEC', 'EXECUTE', 'UNION', '--', '/*', '*/'
    ]
    
    @staticmethod
    def validate(sql: str) -> bool:
        upper_sql = sql.upper()
        
        # 1. 必须是SELECT开头
        if not upper_sql.strip().startswith('SELECT'):
            raise SecurityError("只允许SELECT查询")
        
        # 2. 检查禁止关键字
        for keyword in SQLSecurityFilter.FORBIDDEN_KEYWORDS:
            if keyword in upper_sql:
                raise SecurityError(f"检测到禁止关键字: {keyword}")
        
        # 3. 检查注释
        if '--' in sql or '/*' in sql:
            raise SecurityError("不允许SQL注释")
        
        # 4. 检查多语句
        if ';' in sql and not sql.strip().endswith(';'):
            raise SecurityError("不允许多SQL语句")
        
        return True
    
    @staticmethod
    def sanitize(sql: str) -> str:
        # 强制添加LIMIT防止大数据查询（生产环境可配置）
        if 'LIMIT' not in sql.upper():
            sql = f"{sql.strip()} LIMIT 1000"
        return sql
